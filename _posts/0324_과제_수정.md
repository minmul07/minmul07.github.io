info = "limkyunghee021524293068"
inputdata = input("입력: ")

if (inputdata in info):
    print("입력받은 문자열에 \"limkyunghee021524293068\" 에 포함된 문자가 존재합니다.")
else:
    print("입력받은 문자열에 \"limkyunghee021524293068\" 에 포함된 문자가 존재하지 않습니다.")
